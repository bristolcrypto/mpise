1. support for carry propagation:

   ```
   +=============+===============+==========+==============+=============+
   | instruction | format        | f2       | f3           | opcode      |
   +=============+===============+==========+==============+=============+
   |             |               |          |              |             |
   | cadd        | R4 : 3R  ->1R | 00       | 000          | 0 = 0001011 |
   |             |               |          |              |             |
   +=============+===============+==========+==============+=============+

   R4 : rrrr r00r rrrr rrrr r000 rrrr r000 1011
   #define MATCH_CADD     0x000000B
   #define  MASK_CADD     0x000107B

   .macro cadd     rd, rs1, rs2, rs3             
   .insn r4 CUSTOM_0, 0, 0,                    \rd, \rs1, \rs2, \rs3 
   .endm   

   cadd     rd, rs1, rs2, rs3     {
     x       <- GPR[rs1]
     y       <- GPR[rs2]
     z       <- GPR[rs3]
     r       <- ((x + y) >> 64) + z
     GPR[rd] <- r
   }  

   +=============+===============+==========+==============+=============+
   | instruction | format        | f7       | f3           | opcode      |
   +=============+===============+==========+==============+=============+
   |             |               |          |              |             |
   | sraiadd     | R  : 2R+I->1R | imm      | 001          | 0 = 0001011 |
   |             |               |          |              |             |
   +=============+===============+==========+==============+=============+

   R  : iiii iiir rrrr rrrr r001 rrrr r000 1011
   #define MATCH_SRAIADD  0x000100B
   #define  MASK_SRAIADD  0x000107B

   .macro sraiadd  rd, rs1, rs2,      imm
   .insn r  CUSTOM_0, 1, ( \imm & 0x7F ) << 0, \rd, \rs1, \rs2
   .endm 

   sraiadd  rd, rs1, rs2,     imm {
     x       <- GPR[rs1]
     y       <- GPR[rs2]
     r       <- x + EXTS(y >> imm)
     GPR[rd] <- r
   }
   ```

2. support for    full-radix multiply-add:

   ```
   +=============+===============+==========+==============+=============+
   | instruction | format        | f2       | f3           | opcode      |
   +=============+===============+==========+==============+=============+
   |             |               |          |              |             |
   | maddfrlu    | R4 : 3R  ->1R | 00       | 000          | 1 = 0101011 |
   | maddfrhu    | R4 : 3R  ->1R | 00       | 001          | 1 = 0101011 |
   |             |               |          |              |             |
   +=============+===============+==========+==============+=============+

   R4 : rrrr r00r rrrr rrrr r000 rrrr r000 1011
   #define MATCH_MADDFRLU 0x000002B
   #define  MASK_MADDFRLU 0x000107B
   R4 : rrrr r00r rrrr rrrr r001 rrrr r000 1011
   #define MATCH_MADDFRHU 0x000102B
   #define  MASK_MADDFRHU 0x000107B

   .macro maddfrlu rd, rs1, rs2, rs3            
   .insn r4 CUSTOM_1, 0, 0, \rd, \rs1, \rs2, \rs3 
   .endm                                                     
   .macro maddfrhu rd, rs1, rs2, rs3            
   .insn r4 CUSTOM_1, 1, 0, \rd, \rs1, \rs2, \rs3 
   .endm    

   maddfrlu rd, rs1, rs2, rs3 {   
     x       <- GPR[rs1]
     y       <- GPR[rs2]
     z       <- GPR[rs3]
     m       <- (1 << 64) - 1
     r       <- ( (x * y + z)         & m) 
     GPR[rd] <- r
   }

   maddfrhu rd, rs1, rs2, rs3 {      
     x       <- GPR[rs1]
     y       <- GPR[rs2]
     z       <- GPR[rs3]
     m       <- (1 << 64) - 1
     r       <- (((x * y  + z) >> 64) & m) 
     GPR[rd] <- r
   }
   ```

3. support for reduced-radix multiply-add:

   - common
   
     ```
     f_0(x) = selection of radix based on x
     f_1(x) =  0
     f_2(x) = 64
     f_3(x) = selection of radix based on x
   
     t_maddrrlu x, y, z, c {
       m       <- (1 << f_0(c)) - 1
       r       <- (((x * y) >> f_1(c)) & m) + z 
       return r
     }
      
     t_maddrrlu x, y, z, c {
       m       <- (1 << f_2(c)) - 1
       r       <- (((x * y) >> f_3(c)) & m) + z 
       return r
     }
     ```
   
   - `RV64_ISE_TYPE_1`: not destructive, not stateless (4-bit radix selection immediate)
   
     ```
     +=============+===============+==========+==============+=============+
     | instruction | format        | f2       | f3           | opcode      |
     +=============+===============+==========+==============+=============+
     |             |               |          |              |             |
     | maddrrlu    | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 0 | 2 = 1011011 |
     | maddrrhu    | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 1 | 2 = 1011011 |
     |             |               |          |              |             |
     +=============+===============+==========+==============+=============+

     R4 : rrrr riir rrrr rrrr rii0 rrrr r101 1011
     #define MATCH_MADDFRLU 0x000005B
     #define  MASK_MADDFRLU 0x000107B
     R4 : rrrr riir rrrr rrrr rii1 rrrr r101 1011
     #define MATCH_MADDFRHU 0x000105B
     #define  MASK_MADDFRHU 0x000107B

     .macro maddrrlu rd, rs1, rs2, rs3, imm
     .insn r4 CUSTOM_2, 0 | ( ( \imm & 0x3 ) << 1, ( \imm 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
     .endm
     .macro maddrrhu rd, rs1, rs2, rs3, imm
     .insn r4 CUSTOM_2, 1 | ( ( \imm & 0x3 ) << 1, ( \imm 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
     .endm

     maddrrlu rd, rs1, rs2, rs3      {
       GPR[rd] <- t_maddrrlu( GPR[rs1], GPR[rs2], GPR[rs3], imm         )
     }
      
     maddrrhu rd, rs1, rs2, rs3      {
       GPR[rd] <- t_maddrrhu( GPR[rs1], GPR[rs2], GPR[rs3], imm         )
     }
     ```
   
   - `RV64_ISE_TYPE_2`:     destructive, not stateless (9-bit radix selection immediate)
   
     ```
     +=============+===============+==========+==============+=============+
     | instruction | format        | f7       | f3           | opcode      |
     +=============+===============+==========+==============+=============+
     |             |               |          |              |             |
     | maddrrlu    | R  : 2R+I->1R | imm[6:0] | imm[1:0] | 0 | 2 = 1011011 |
     | maddrrhu    | R  : 2R+I->1R | imm[6:0] | imm[1:0] | 1 | 2 = 1011011 |
     |             |               |          |              |             |
     +=============+===============+==========+==============+=============+

     R  : iiii iiir rrrr rrrr rii0 rrrr r101 1011
     #define MATCH_MADDFRLU 0x000005B
     #define  MASK_MADDFRLU 0x000107B
     R  : iiii iiir rrrr rrrr rii1 rrrr r101 1011
     #define MATCH_MADDFRHU 0x000105B
     #define  MASK_MADDFRHU 0x000107B
   
     .macro maddrrlu rd, rs1, rs2, rs3, imm
     .insn r  CUSTOM_2, 0 | ( ( \imm & 0x003 ) << 1, ( \imm 0x1FC ) >> 2, \rd, \rs1, \rs2, \rs3
     .endm
     .macro maddrrhu rd, rs1, rs2, rs3, imm
     .insn r  CUSTOM_2, 1 | ( ( \imm & 0x003 ) << 1, ( \imm 0x1FC ) >> 2, \rd, \rs1, \rs2, \rs3
     .endm

     maddrrlu rd, rs1, rs2           {
       GPR[rd] <- t_maddrrlu( GPR[rs1], GPR[rs2], GPR[rd],  imm         )
     }
      
     maddrrhu rd, rs1, rs2           {
       GPR[rd] <- t_maddrrhu( GPR[rs1], GPR[rs2], GPR[rd],  imm         )
     }
     ```
