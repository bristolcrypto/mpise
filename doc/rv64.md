# 1. support for carry propagation.

- generic.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | sraiadd     | R  : 2R+I->1R | imm      | 001          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R  : iiii iiir rrrr rrrr r001 rrrr r000 1011
  #define MATCH_SRAIADD  0x000100B
  #define  MASK_SRAIADD  0x000107B

  .macro sraiadd  rd, rs1, rs2, imm
  .insn r  CUSTOM_0, 1, ( \imm & 0x7F ) << 0, \rd, \rs1, \rs2
  .endm 

  sraiadd rd, rs1, rs2, imm {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    r       <- x + EXTS(y >> imm)
    GPR[rd] <- r
  }
  ```

- only if `( MPISE_DESTRUCTIVE == 0 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | cacc        | R4 : 3R  ->1R | 00       | 000          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R4 : rrrr r00r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_CACC     0x000000B
  #define  MASK_CACC     0x000107B
   
  .macro cacc rd, rs1, rs2, rs3             
  .insn r4 CUSTOM_0, 0, 0, \rd, \rs1, \rs2, \rs3 
  .endm
   
  cacc rd, rs1, rs2, rs3 {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rs3]
    r       <- ((x + y) >> 64) + z
    GPR[rd] <- r
  }  
  ```

- only if `( MPISE_DESTRUCTIVE == 1 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | cacc        | R  : 2R  ->1R | 0000000  | 000          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R  : 0000 000r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_CACC     0x000000B
  #define  MASK_CACC     0x000107B
   
  .macro cacc rd, rs1, rs2
  .insn r  CUSTOM_0, 0, 0, \rd, \rs1, \rs2
  .endm
   
  cacc rd, rs1, rs2, rs3 {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rd ]
    r       <- ((x + y) >> 64) + z
    GPR[rd] <- r
  }  
  ```

# 2. support for    full-radix multiply-add.

- only if `( MPISE_DESTRUCTIVE == 0 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccfrlu    | R4 : 3R  ->1R | 00       | 000          | 1 = 0101011 |
  | maccfrhu    | R4 : 3R  ->1R | 00       | 001          | 1 = 0101011 |
  +=============+===============+==========+==============+=============+
   
  R4 : rrrr r00r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_MACCFRLU 0x000002B
  #define  MASK_MACCFRLU 0x000107B
  R4 : rrrr r00r rrrr rrrr r001 rrrr r000 1011
  #define MATCH_MACCFRHU 0x000102B
  #define  MASK_MACCFRHU 0x000107B
   
  .macro maccfrlu rd, rs1, rs2, rs3            
  .insn r4 CUSTOM_1, 0, 0, \rd, \rs1, \rs2, \rs3 
  .endm                                                     
  .macro maccfrhu rd, rs1, rs2, rs3            
  .insn r4 CUSTOM_1, 1, 0, \rd, \rs1, \rs2, \rs3 
  .endm    
   
  maccfrlu rd, rs1, rs2, rs3 {   
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rs3]
    m       <- (1 << 64) - 1
    r       <- ( (x * y + z)         & m) 
    GPR[rd] <- r
  }
   
  maccfrhu rd, rs1, rs2, rs3 {      
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rs3]
    m       <- (1 << 64) - 1
    r       <- (((x * y  + z) >> 64) & m) 
    GPR[rd] <- r
  }
  ```

- only if `( MPISE_DESTRUCTIVE == 1 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccfrlu    | R  : 2R  ->1R | 0000000  | 000          | 1 = 0101011 |
  | maccfrhu    | R  : 2R  ->1R | 0000000  | 001          | 1 = 0101011 |
  +=============+===============+==========+==============+=============+
   
  R  : 0000 000r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_MACCFRLU 0x000002B
  #define  MASK_MACCFRLU 0x000107B
  R  : 0000 000r rrrr rrrr r001 rrrr r000 1011
  #define MATCH_MACCFRHU 0x000102B
  #define  MASK_MACCFRHU 0x000107B
   
  .macro maccfrlu rd, rs1, rs2, rs3            
  .insn r  CUSTOM_1, 0, 0, \rd, \rs1, \rs2
  .endm
  .macro maccfrhu rd, rs1, rs2, rs3            
  .insn r  CUSTOM_1, 1, 0, \rd, \rs1, \rs2
  .endm
   
  maccfrlu rd, rs1, rs2, rs3 {   
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rs3]
    m       <- (1 << 64) - 1
  ```

# 3. support for reduced-radix multiply-add.

- We reserve a placeholder CSR to support stateful variants of the ISE,
  selecting the first available unprivileged, custom read/write CSR:
  this is at address `0x800`.

- only if `( MPISE_DESTRUCTIVE == 0 )` and `( MPISE_STATELESS == 0 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccrrlu    | R4 : 3R  ->1R | 0        |            0 | 2 = 1011011 |
  | maccrrhu    | R4 : 3R  ->1R | 0        |            1 | 2 = 1011011 |
  +=============+===============+==========+==============+=============+

  R4 : rrrr riir rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCRRLU 0x000005B
  #define  MASK_MACCRRLU 0x000107B
  R4 : rrrr riir rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCRRHU 0x000105B
  #define  MASK_MACCRRHU 0x000107B

  .macro maccrrlu rd, rs1, rs2, rs3
  .insn r4 CUSTOM_2, 0, 0, \rd, \rs1, \rs2, \rs3
  .endm
  .macro maccrrhu rd, rs1, rs2, rs3
  .insn r4 CUSTOM_2, 1, 0  \rd, \rs1, \rs2, \rs3
  .endm

  maccrrlu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_maccrrlu( GPR[rs1], GPR[rs2], GPR[rs3], CSR[macc] )
  }
      
  maccrrhu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_maccrrhu( GPR[rs1], GPR[rs2], GPR[rs3], CSR[macc] )
  }   
  ```

- only if `( MPISE_DESTRUCTIVE == 0 )` and `( MPISE_STATELESS == 1 )` (4-bit radix selection immediate).
   
  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccrrlu    | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 0 | 2 = 1011011 |
  | maccrrhu    | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 1 | 2 = 1011011 |
  +=============+===============+==========+==============+=============+

  R4 : rrrr riir rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCRRLU 0x000005B
  #define  MASK_MACCRRLU 0x000107B
  R4 : rrrr riir rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCRRHU 0x000105B
  #define  MASK_MACCRRHU 0x000107B

  .macro maccrrlu rd, rs1, rs2, rs3, imm
  .insn r4 CUSTOM_2, 0 | ( ( \imm & 0x3 ) << 1, ( \imm 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
  .endm
  .macro maccrrhu rd, rs1, rs2, rs3, imm
  .insn r4 CUSTOM_2, 1 | ( ( \imm & 0x3 ) << 1, ( \imm 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
  .endm

  maccrrlu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_maccrrlu( GPR[rs1], GPR[rs2], GPR[rs3], imm )
  }
      
  maccrrhu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_maccrrhu( GPR[rs1], GPR[rs2], GPR[rs3], imm )
  }
  ```

- only if `( MPISE_DESTRUCTIVE == 1 )` and `( MPISE_STATELESS == 0 )`.   
   
  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccrrlu    | R  : 2R  ->1R | 0        |            0 | 2 = 1011011 |
  | maccrrhu    | R  : 2R  ->1R | 0        |            1 | 2 = 1011011 |
  +=============+===============+==========+==============+=============+

  R  : iiii iiir rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCRRLU 0x000005B
  #define  MASK_MACCRRLU 0x000107B
  R  : iiii iiir rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCRRHU 0x000105B
  #define  MASK_MACCRRHU 0x000107B

  .macro maccrrlu rd, rs1, rs2
  .insn r  CUSTOM_2, 0, 0, \rd, \rs1, \rs2
  .endm
  .macro maccrrhu rd, rs1, rs2
  .insn r  CUSTOM_2, 1, 0, \rd, \rs1, \rs2
  .endm

  maccrrlu rd, rs1, rs2 {
    GPR[rd] <- t_maccrrlu( GPR[rs1], GPR[rs2], GPR[rd],  CSR[macc] )
  }
      
  maccrrhu rd, rs1, rs2 {
    GPR[rd] <- t_maccrrhu( GPR[rs1], GPR[rs2], GPR[rd],  CSR[macc] )
  }   
  ```

- only if `( MPISE_DESTRUCTIVE == 1 )` and `( MPISE_STATELESS == 1 )` (9-bit radix selection immediate).
   
  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | maccrrlu    | R  : 2R+I->1R | imm[6:0] | imm[1:0] | 0 | 2 = 1011011 |
  | maccrrhu    | R  : 2R+I->1R | imm[6:0] | imm[1:0] | 1 | 2 = 1011011 |
  +=============+===============+==========+==============+=============+

  R  : iiii iiir rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCRRLU 0x000005B
  #define  MASK_MACCRRLU 0x000107B
  R  : iiii iiir rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCRRHU 0x000105B
  #define  MASK_MACCRRHU 0x000107B
   
  .macro maccrrlu rd, rs1, rs2, imm
  .insn r  CUSTOM_2, 0 | ( ( \imm & 0x003 ) << 1, ( \imm 0x1FC ) >> 2, \rd, \rs1, \rs2
  .endm
  .macro maccrrhu rd, rs1, rs2, imm
  .insn r  CUSTOM_2, 1 | ( ( \imm & 0x003 ) << 1, ( \imm 0x1FC ) >> 2, \rd, \rs1, \rs2
  .endm

  maccrrlu rd, rs1, rs2, imm {
    GPR[rd] <- t_maccrrlu( GPR[rs1], GPR[rs2], GPR[rd],  imm )
  }
      
  maccrrhu rd, rs1, rs2, imm {
    GPR[rd] <- t_maccrrhu( GPR[rs1], GPR[rs2], GPR[rd],  imm )
  }
  ```

- generic.
   
  ```
  f_0(x) = selection of radix based on x
  f_1(x) =  0
  f_2(x) = 64
  f_3(x) = selection of radix based on x

  t_maccrrlu x, y, z, c {
    m       <- (1 << f_0(c)) - 1
    r       <- (((x * y) >> f_1(c)) & m) + z 
    return r
  }
      
  t_maccrrlu x, y, z, c {
    m       <- (1 << f_2(c)) - 1
    r       <- (((x * y) >> f_3(c)) & m) + z 
    return r
  }
  ```
