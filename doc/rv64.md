# 1. support for carry propagation.

- generic.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | sraiadd     | R  : 2R+I->1R | imm      | 001          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R  : iiii iiir rrrr rrrr r001 rrrr r000 1011
  #define MATCH_SRAIADD  0x000100B
  #define  MASK_SRAIADD  0x000707F

  .macro sraiadd  rd, rs1, rs2, imm
  .insn r  CUSTOM_0, 1, ( \imm & 0x7F ) << 0, \rd, \rs1, \rs2
  .endm 

  sraiadd rd, rs1, rs2, imm {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    r       <- x + EXTS(y >> imm)
    GPR[rd] <- r
  }
  ```

- only if `( MPISE_DESTRUCTIVE == 0 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | cacc        | R4 : 3R  ->1R | 00       | 000          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R4 : rrrr r00r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_CACC     0x000000B
  #define  MASK_CACC     0x000707F
   
  .macro cacc rd, rs1, rs2, rs3             
  .insn r4 CUSTOM_0, 0, 0, \rd, \rs1, \rs2, \rs3 
  .endm
   
  cacc rd, rs1, rs2, rs3 {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rs3]
    r       <- ((x + y) >> w) + z
    GPR[rd] <- r
  }  
  ```

- only if `( MPISE_DESTRUCTIVE == 1 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f7       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | cacc        | R  : 2R  ->1R | 0000000  | 000          | 0 = 0001011 |
  +=============+===============+==========+==============+=============+
   
  R  : 0000 000r rrrr rrrr r000 rrrr r000 1011
  #define MATCH_CACC     0x000000B
  #define  MASK_CACC     0x000707F
   
  .macro cacc rd, rs1, rs2
  .insn r  CUSTOM_0, 0, 0, \rd, \rs1, \rs2
  .endm
   
  cacc rd, rs1, rs2, rs3 {
    x       <- GPR[rs1]
    y       <- GPR[rs2]
    z       <- GPR[rd ]
    r       <- ((x + y) >> 64) + z
    GPR[rd] <- r
  }  
  ```

# 2. support for multiply-add (for full- and reduced-radix).

- We reserve a placeholder CSR to support stateful variants of the ISE,
  selecting the first available unprivileged, custom read/write CSR:
  this is at address `0x800`.

- Some variants permit a larger radix selector than others.  Simply to 
  make the (prototype) design and implementation more uniform, we will
  assume a 4-bit selector 
  `0 <= c < 2^4 = 16`
  for all variants.  Having done so, we then assume
 
  ```
  c == 15 => use    full-radix                               
          => implied radix is `r = 64`

  c != 15 => use reduced-radix with implied radix of `(w-15)+c`
          => implied radix is `49 <= r < 64`
  ```

- generic.
   
  ```
  t_macclu x, y, z, c {
    t_0 <- x * y
    t_1 <- f_0(c) ? z : 0
    t_2 <- f_0(c) ? 0 : z
    m   <- (1 << f_1(c)) - 1
    r   <- (((t_0 + t_1) >> f_2(c)) & m) + t_2
    return r
  }
  ```

  and

  ```
  t_macchi x, y, z, c {
    t_0 <- x * y
    t_1 <- f_3(c) ? z : 0
    t_2 <- f_3(c) ? 0 : z
    m   <- (1 << f_4(c)) - 1
    r   <- (((t_0 + t_1) >> f_5(c)) & m) + t_2
    return r
  }
  ```

  where

  ```
  f_0(c) =  0       if c != 15 (i.e., c means reduced-radix)
            1       if c == 15 (i.e., c means    full-radix)

  f_1(c) = (w-15)+c if c != 15 (i.e., c means reduced-radix)
            w       if c == 15 (i.e., c means    full-radix)

  f_2(c) =  0       if c != 15 (i.e., c means reduced-radix)
            0       if c == 15 (i.e., c means    full-radix)

  f_3(c) =  0       if c != 15 (i.e., c means reduced-radix)
            1       if c == 15 (i.e., c means    full-radix)

  f_4(c) =  w       if c != 15 (i.e., c means reduced-radix)
            w       if c == 15 (i.e., c means    full-radix)

  f_5(c) = (w-15)+c if c != 15 (i.e., c means reduced-radix)
            w       if c == 15 (i.e., c means    full-radix)
  ```

- only if `( MPISE_DESTRUCTIVE == 0 )` and `( MPISE_STATELESS == 0 )`.

  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | macclu      | R4 : 3R  ->1R | 0        |            0 | 1 = 0101011 |
  | macchu      | R4 : 3R  ->1R | 0        |            1 | 1 = 0101011 |
  +=============+===============+==========+==============+=============+

  R4 : rrrr r00r rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCLU 0x000002B
  #define  MASK_MACCLU 0x000107F
  R4 : rrrr r00r rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCHU 0x000102B
  #define  MASK_MACCHU 0x000107F

  .macro macclu rd, rs1, rs2, rs3
  .insn r4 CUSTOM_1, 0, 0, \rd, \rs1, \rs2, \rs3
  .endm
  .macro macchu rd, rs1, rs2, rs3
  .insn r4 CUSTOM_1, 1, 0  \rd, \rs1, \rs2, \rs3
  .endm

  macclu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_macclu( GPR[rs1], GPR[rs2], GPR[rs3], CSR[macc] )
  }
      
  macchu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_macchu( GPR[rs1], GPR[rs2], GPR[rs3], CSR[macc] )
  }   
  ```

- only if `( MPISE_DESTRUCTIVE == 0 )` and `( MPISE_STATELESS == 1 )`
   
  ```
  +=============+===============+==========+==============+=============+
  | instruction | format        | f2       | f3           | opcode      |
  +=============+===============+==========+==============+=============+
  | macclu      | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 0 | 1 = 0101011 |
  | macchu      | R4 : 3R+I->1R | imm[3:2] | imm[1:0] | 1 | 1 = 0101011 |
  +=============+===============+==========+==============+=============+

  R4 : rrrr riir rrrr rrrr rii0 rrrr r101 1011
  #define MATCH_MACCLU 0x000002B
  #define  MASK_MACCLU 0x000107F
  R4 : rrrr riir rrrr rrrr rii1 rrrr r101 1011
  #define MATCH_MACCHU 0x000102B
  #define  MASK_MACCHU 0x000107F

  .macro macclu rd, rs1, rs2, rs3, imm
  .insn r4 CUSTOM_1, 0 | ( ( \imm & 0x3 ) << 1 ), ( \imm & 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
  .endm
  .macro macchu rd, rs1, rs2, rs3, imm
  .insn r4 CUSTOM_1, 1 | ( ( \imm & 0x3 ) << 1 ), ( \imm & 0xC ) >> 2, \rd, \rs1, \rs2, \rs3
  .endm

  macclu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_macclu( GPR[rs1], GPR[rs2], GPR[rs3], imm )
  }
      
  macchu rd, rs1, rs2, rs3 {
    GPR[rd] <- t_macchu( GPR[rs1], GPR[rs2], GPR[rs3], imm )
  }
  ```
